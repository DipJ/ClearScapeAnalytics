
CREATE TABLE "Customer" (
	"CustomerID" BIGINT NOT NULL, 
	"FirstName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"LastName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"DateOfBirth" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Gender" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Address" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"City" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"LocalityID" BIGINT NOT NULL, 
	"Country" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Email" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"PhoneNumber" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"AccountNumber" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"AccountType" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Balance" FLOAT NOT NULL, 
	"BranchID" BIGINT NOT NULL, 
	"AccountStatus" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"CreditScore" BIGINT NOT NULL, 
	"Income" FLOAT NOT NULL
)


CREATE TABLE "Interest" (
	"ID" BIGINT NOT NULL, 
	"MinCreditScore" BIGINT NOT NULL, 
	"MaxCreditScore" BIGINT NOT NULL, 
	"InterestRate" FLOAT NOT NULL
)


CREATE TABLE "Locality" (
	"LocalityID" BIGINT NOT NULL, 
	"LocalityName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"LocalityDescription" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"DistanceFromAirport" BIGINT NOT NULL, 
	"LivabilityScore" FLOAT NOT NULL
)


CREATE TABLE "NearbyLocality" (
	"LocalityID" BIGINT NOT NULL, 
	"LocalityName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"NearbyLocalityID" BIGINT NOT NULL
)


CREATE TABLE "RealEstate" (
	"LocalityID" BIGINT NOT NULL, 
	"PropertyID" BIGINT NOT NULL, 
	"PropertyType" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"BuildingSize" BIGINT NOT NULL, 
	"LandSize" BIGINT NOT NULL, 
	"PreferredSize" BIGINT NOT NULL, 
	"OpenDate" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"ListingAgency" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Price" FLOAT NOT NULL, 
	"Address" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"City" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"State" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"ZipCode" BIGINT NOT NULL, 
	"Phone" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"BedroomCount" BIGINT NOT NULL, 
	"BathroomCount" BIGINT NOT NULL, 
	"ParkingCount" BIGINT NOT NULL
)


CREATE TABLE "School" (
	"SchoolID" BIGINT NOT NULL, 
	"SchoolName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"LocalityID" BIGINT NOT NULL
)


CREATE TABLE "ShoppingCenter" (
	"StoreID" BIGINT NOT NULL, 
	"StoreName" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"Type" VARCHAR(1024) CHAR SET UNICODE NOT NULL, 
	"LocalityID" BIGINT NOT NULL
)
